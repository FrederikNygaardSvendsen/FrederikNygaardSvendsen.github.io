# Create branch
    
    git branch someBranch

# switch branch:

    git checkout someBranch
    
# show branches:

    git branch
    
# delete branch:

    git branch -d branchToDelete
    
# Merge other branch into current branch

    git merge branchToMergeIntoCurrentBranch
